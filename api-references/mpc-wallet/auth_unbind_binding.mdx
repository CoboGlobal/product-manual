---
title: Unbind Binding
api: POST /v1/custody/guard/unbind_binding/
contentType: multipart/form-data
description: "This endpoint allows users to unbind their broker account from their MPC Wallet account. It requires a POST request with a 'user_id' field in the request body. The response is a JSON object containing a 'statement_id' field which can be used to query the status of the unbinding process."
---

<ParamField body="user_id" type="String" required>unique user ID</ParamField>

   <ResponseField name="result" type="object">

           <ResponseField name="statement_id" type="String" >unique auth statement id</ResponseField>
      </Expandable>
</ResponseField>
