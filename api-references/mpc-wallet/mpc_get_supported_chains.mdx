---
title: Get Supported Chains
api: GET /v1/custody/mpc/get_supported_chains/ 
description: "This endpoint returns a JSON response with a 'chain_codes' field containing a list of supported chain codes for the MPC custody wallet associated with the authenticated user."
---
#### Request 
None

<ResponseField name="success" type="bool">request successful or failed</ResponseField>
   <ResponseField name="result" type="object">
      <Expandable title="object">
        <ResponseField name="chain_codes" type="String[ ]" >List of supported chain codes.</ResponseField>
      </Expandable>
</ResponseField>
